# ── LLM Providers ─────────────────────────────────────────────────────────────
google-genai            # Current provider (works with UPI) — new SDK
anthropic               # Future provider (add API key when available)
openai                  # Alternative provider
ollama                  # Local model client (agents run locally)

# ── Speech I/O ────────────────────────────────────────────────────────────────
groq                    # Groq API client — Whisper STT
sounddevice             # Microphone capture
numpy                   # Audio array processing
pynput                  # Keyboard listener for push-to-talk
kokoro-onnx             # Primary TTS engine (high quality, local)
soundfile               # Write WAV files for Kokoro output

# ── Web Agent ─────────────────────────────────────────────────────────────────
duckduckgo-search       # Web search (no API key required)
requests                # HTTP requests for page fetching
beautifulsoup4          # HTML parsing and text extraction

# ── Utilities ─────────────────────────────────────────────────────────────────
python-dotenv           # Load .env file for API keys
